// Processing pipeline สำหรับ uploaded documents
async function processUploadedDocument(shopId: string, file: File) {
  let extractedText = ''

  // 1. Extract ข้อความตามประเภทไฟล์
  switch (file.type) {
    case 'application/pdf':
      extractedText = await extractPDF(file)          // pdf-parse
      break
    case 'application/vnd.openxmlformats...':
      extractedText = await extractDocx(file)         // mammoth
      break
    case 'text/csv':
      extractedText = await extractCSV(file)          // papaparse
      break
    case 'image/png':
    case 'image/jpeg':
      // Gemini Vision อ่านเมนูจากรูปภาพ
      extractedText = await gemini.generateContent({
        contents: [{ parts: [{ inlineData: { data: file, mimeType: file.type } }] }],
        systemInstruction: 'อ่านและสกัดข้อมูลเมนู ราคา และรายละเอียดทั้งหมดจากภาพนี้เป็น structured text'
      })
      break
  }

  // 2. ให้ AI แยกเป็น knowledge documents
  const documents = await splitIntoKnowledgeDocs(shopId, extractedText)

  // 3. Preview ให้เจ้าของร้านตรวจก่อน confirm
  return { documents, previewText: extractedText }
}
```

// Style profile → inject ใน system prompt
interface ShopStyleProfile {
  botName: string                    // "น้องปัญญา"
  formalityLevel: number             // 0-100 (0=ทางการ, 100=เป็นกันเอง)
  emojiUsage: 'none' | 'some' | 'lots'
  replyLength: 'short' | 'medium' | 'long'
  language: 'thai' | 'mixed' | 'english'
  closingPhrase: string
  openingGreeting: string
  guardrails: string[]               // สิ่งที่ห้ามพูด
}

function styleProfileToPrompt(style: ShopStyleProfile): string {
  return `
    ชื่อของคุณคือ "${style.botName}"

    สไตล์การตอบ:
    - น้ำเสียง: ${style.formalityLevel > 60 ? 'เป็นกันเอง สบายๆ' : 'สุภาพ เป็นมืออาชีพ'}
    - Emoji: ${style.emojiUsage === 'none' ? 'ไม่ใช้ emoji' : style.emojiUsage === 'some' ? 'ใช้ emoji พอประมาณ 1-2 ตัวต่อข้อความ' : 'ใช้ emoji ได้เยอะ สร้างความสนุก'}
    - ความยาว: ${style.replyLength === 'short' ? '1-2 ประโยคเท่านั้น' : style.replyLength === 'medium' ? '2-4 ประโยค' : '3-6 ประโยค รายละเอียดครบ'}
    - ปิดท้ายทุกข้อความด้วย: "${style.closingPhrase}"
    - ทักทายแรกด้วย: "${style.openingGreeting}"

    ห้ามพูดถึงสิ่งเหล่านี้เด็ดขาด:
    ${style.guardrails.map(g => `- ${g}`).join('\n')}
  `
}

async function handleReplyCorrection(

  correction: {
    originalReply: string
    correctedReply: string
    customerMessage: string
    saveAsExample: boolean
    addToKnowledge: boolean
    newKnowledgeText?: string
  }
) {
  // 1. บันทึก corrected reply ส่งกลับให้ลูกค้า

  if (correction.saveAsExample) {

      customerMessage: correction.customerMessage,
      aiDraftReply: correction.originalReply,
      finalReply: correction.correctedReply,

    })
  }

  if (correction.addToKnowledge && correction.newKnowledgeText) {
    // 3. เพิ่มเข้า knowledge base (Layer 1)
    await saveFreeformKnowledge(shopId, correction.newKnowledgeText)
  }

  // 4. อัปเดต score ของ pattern ที่ทำให้ AI ตอบผิด (ลด weight)
  await penalizePattern(shopId, correction.originalReply)
}
```
